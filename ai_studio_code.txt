import React, { useState, useEffect } from 'react';
import { Game } from '../types';
import { ContentAPI } from '../lib/content-api';
import { 
  Gamepad2, 
  Search, 
  Filter, 
  Headphones, 
  Tv, 
  LayoutGrid, 
  Loader2,
  CheckCircle2,
  Disc
} from 'lucide-react';

interface GameRoomProps {
  activeGame: Game;
  onSelectGame: (game: Game) => void;
}

const CATEGORIES = [
  { id: 'All', icon: LayoutGrid },
  { id: 'Arcade', icon: Gamepad2 },
  { id: 'Zenner', icon: Headphones },
  { id: 'Media', icon: Tv },
];

export const GameRoom: React.FC<GameRoomProps> = ({ activeGame, onSelectGame }) => {
  const [games, setGames] = useState<Game[]>([]);
  const [isLoading, setIsLoading] = useState(true);
  const [selectedCategory, setSelectedCategory] = useState('All');
  const [searchQuery, setSearchQuery] = useState('');

  useEffect(() => {
    loadGames();
  }, [selectedCategory]);

  useEffect(() => {
    if (searchQuery.trim() === '') {
      loadGames();
    } else {
      const performSearch = async () => {
        setIsLoading(true);
        const results = await ContentAPI.search(searchQuery);
        const filtered = selectedCategory === 'All' 
          ? results 
          : results.filter(g => g.category === selectedCategory);
        setGames(filtered);
        setIsLoading(false);
      };
      const timer = setTimeout(performSearch, 500);
      return () => clearTimeout(timer);
    }
  }, [searchQuery]);

  const loadGames = async () => {
    setIsLoading(true);
    try {
      const data = await ContentAPI.fetchByCategory(selectedCategory);
      setGames(data);
    } catch (e) {
      console.error("Failed to load content", e);
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="space-y-6 animate-in fade-in h-full flex flex-col">
      
      <div className="flex flex-col gap-4">
        <div className="flex items-center justify-between">
          <div>
            <h2 className="text-2xl font-display font-bold text-white tracking-wide">THE ARMORY</h2>
            <p className="text-xs font-mono text-slate-400">SELECT DATA CARTRIDGE FOR MINING SHELL</p>
          </div>
        </div>

        <div className="relative w-full">
           <Search className="absolute left-3 top-1/2 transform -translate-y-1/2 text-slate-500" size={18} />
           <input 
             type="text" 
             placeholder="Search by name, hash, or tags..." 
             value={searchQuery}
             onChange={(e) => setSearchQuery(e.target.value)}
             className="w-full bg-slate-900 border border-slate-700 rounded-lg py-3 pl-10 pr-4 text-white focus:outline-none focus:border-blue-500 font-mono text-sm transition-colors"
           />
        </div>

        <div className="flex gap-2 overflow-x-auto pb-2 scrollbar-hide">
           {CATEGORIES.map(cat => (
             <button
               key={cat.id}
               onClick={() => { setSelectedCategory(cat.id); setSearchQuery(''); }}
               className={`
                 flex items-center gap-2 px-4 py-2 rounded-full text-xs font-bold font-display uppercase tracking-wider whitespace-nowrap border transition-all
                 ${selectedCategory === cat.id 
                   ? 'bg-blue-500 text-black border-blue-500 shadow-[0_0_15px_rgba(93,173,226,0.4)]' 
                   : 'bg-slate-900 text-slate-400 border-slate-700 hover:border-slate-500'
                 }
               `}
             >
               <cat.icon size={14} />
               {cat.id}
             </button>
           ))}
        </div>
      </div>

      <div className="flex-1 overflow-y-auto pr-2 pb-20">
        {isLoading ? (
           <div className="h-64 flex flex-col items-center justify-center text-blue-500">
              <Loader2 size={40} className="animate-spin mb-4" />
              <p className="font-mono text-xs animate-pulse">ESTABLISHING UPLINK...</p>
           </div>
        ) : games.length === 0 ? (
           <div className="h-64 flex flex-col items-center justify-center text-slate-600">
              <Disc size={40} className="mb-4 opacity-50" />
              <p className="font-mono text-xs">NO CARTRIDGES FOUND</p>
           </div>
        ) : (
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            {games.map(game => {
              const isActive = activeGame.id === game.id;
              return (
                <button 
                  key={game.id}
                  onClick={() => onSelectGame(game)}
                  className={`
                    group relative overflow-hidden rounded-xl border transition-all text-left p-0 h-40 flex flex-row
                    ${isActive 
                      ? 'bg-slate-800 border-blue-500 shadow-[0_0_20px_rgba(93,173,226,0.2)]' 
                      : 'bg-slate-900 border-slate-700 hover:border-slate-500 hover:bg-slate-800'
                    }
                  `}
                >
                  <div className={`w-2 h-full bg-blue-500 group-hover:w-3 transition-all opacity-50 group-hover:opacity-100`}></div>
                  
                  <div className="p-4 flex-1 flex flex-col justify-between relative z-10">
                    <div>
                       <div className="flex justify-between items-start">
                          <h3 className={`font-bold text-lg leading-none ${isActive ? 'text-blue-400' : 'text-white'} group-hover:text-blue-400 transition-colors`}>
                            {game.name}
                          </h3>
                          {isActive && (
                            <div className="bg-blue-500/20 text-blue-400 p-1 rounded-full animate-pulse">
                              <CheckCircle2 size={16} />
                            </div>
                          )}
                       </div>
                       <p className="text-[10px] text-slate-500 font-mono mt-1 uppercase tracking-widest">{game.category}</p>
                       <p className="text-xs text-slate-400 mt-2 line-clamp-2">{game.description}</p>
                    </div>

                    <div className="flex items-center gap-2 pt-3 border-t border-white/5">
                        <span className="text-2xl">{game.thumbnail}</span>
                        <div className="flex flex-col">
                           <span className="text-[9px] text-slate-600 font-mono">HASH VERIFIED</span>
                           <span className="text-[9px] text-slate-500 font-mono tracking-tighter">{game.hash.substring(0, 16)}...</span>
                        </div>
                    </div>
                  </div>

                  {isActive && (
                     <div className="absolute inset-0 border-2 border-blue-500 rounded-xl pointer-events-none animate-pulse"></div>
                  )}
                </button>
              );
            })}
          </div>
        )}
      </div>
    </div>
  );
};
